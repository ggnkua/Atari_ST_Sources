long devconnect( sourcedevice, destinationdevice, clock, prescale, 
protokol)

sourcedevice:     3 = Microfoon
                  2 = Externe input
                  1 = DSP zend
                  0 = DMA play

Destinationdevice:0x8 = Hoofdtelefoon
                  0x4 = Externe uitgang
                  0x2 = DSP ontvang
                  0x1 = DMA record

Clock: 0 = interne klok 25.175 MHZ
       1 = Externe klok
       2 = interne klok 32 MHZ

prescale:  Waarde tussen 1 en 12, die de sample rate aan geeft.

           1 = 50 KHZ
           2 = 33 KHZ
           3 = 25 KHZ
           4 = 20 KHZ
           5 = 16 KHZ
           7 = 12 KHZ
           9 = 10 KHZ
          11 =  8 KHZ

protokol:  0 = Handshaking
           1 = Geen handshaking
