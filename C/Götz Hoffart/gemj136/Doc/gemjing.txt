     msg[4] = (WORD)((LONG)args & 0x0000ffff);

     msg[5] = 0;
     msg[6] = 0;
     msg[7] = 0;

     /* Die VA_START-Nachricht verschicken */
     appl_write(gemjing_id, 16, msg);
   }
   return ;
}
