   This file contains the source code for a complete ecoding/decoding
   package and sample drivers. 

   Based on "Data Compression with Huffman Coding," BYTE March 1986.

   Copyright 1986 by Jonathan Amsterdam.  All Rights Reserved.

   Modified from the original MacModula-2 to TDI Modula-2 for Atari ST

  There are eight modules inlcuded :

  BitStream.DEF       BitStream.MOD
  Myterminal.DEF      MyTerminal.MOD
  Huffman.DEF         Huffman.MOD

  Compress.MOD & UnCompress.MOD

  Each module is separated by a (*---...---*) line for easy identification

  The modules must first be separated into files ( with the Write... 
  command of the editor ).

  MyTerminal should be compiled first, followed by Bitream, Huffman,
  Compress and UnCompress.

  I got the original code as posted by Jonathan Amsterdam on COMPUSERVE
  where he put it into the public domain.  

  Feel free to distribute in the PUBLIC DOMAIN.

  I made extensive modification to the original code.  Making it run on
  the ST with TDI Modula-2 only took a couple of hours - the only 
  differences being the character code handling, the Open/CloseStream
  procedures and the need to pass open arrays as VAR.

  I changed the BitStream module so that it writes and reads words,
  most significant bit first, least last.  I also replaced Read/WriteCard
  with Read/WriteLongCard so that larger files can be correctly handled 
  and reported on.  I replaced two of his modules CharStream and
  StringStuff with the standard equivilents and I removed extraneous
  procedures from MyTerminal. 

  Most of the other changes were formatting and naming.

IMPLEMENTATION MODULE BitStream;

(* File compression algorithm using Huffman coding.
   Based on "Data Compression with Huffman Coding," BYTE March 1986.

   Copyright 1986 by Jonathan Amsterdam.  All Rights Reserved.

   Modified from the original MacModula-2 to TDI Modula-2 for Atari ST

   by Ken Westerback ( 73547,3520 ) August 15, 1986

FROM Streams IMPORT 
  (* type *) Stream, StreamKinds, 
  (* proc *) OpenStream, CloseStream, Read16Bit, Write16Bit;

IMPORT Streams;   (* get Streams.EOS as qualified identifier *)

FROM Storage IMPORT 
  (* proc *) ALLOCATE, DEALLOCATE;

FROM MyTerminal IMPORT 
  (* proc *) Fatal;


CONST highBit = 15;  (* highest numbered bit in a BITSET *)
      lowBit  =  0;  (* lowest ...                       *)


TYPE wordRange = [ lowBit..highBit];

     charRange = [ lowBit..7 ];

     bitStream = POINTER TO bsRec;

         bsRec = RECORD
                 stream     : Stream;
                 streamType : StreamKinds;
                 curWord    : BITSET;
                 curBit     : wordRange;
                 END;


PROCEDURE OpenBitStream
  ( VAR         bs : bitStream;     (* in/out *)
    VAR   fileName : ARRAY OF CHAR; (* in     *) 
        streamType : StreamKinds;   (* in     *)
    VAR        err : INTEGER        (* out    *) );

  BEGIN
  
  NEW(bs);
 
  bs^.streamType := streamType;

  OpenStream ( bs^.stream, fileName, streamType, err );

  IF err # 0 THEN (* no connection *)
    RETURN

  ELSIF streamType = READ THEN
    bs^.curBit := lowBit;

  ELSE
    bs^.curBit := highBit;
    bs^.curWord := {};

  END;

  END OpenBitStream;


PROCEDURE Disconnect
  ( bs : bitStream (* in/out *) );

  VAR err : INTEGER;

  BEGIN

  IF ( bs^.streamType = READWRITE ) AND ( bs^.curBit # highBit ) THEN  
    (* flush the last word *)
    Write16Bit ( bs^.stream, bs^.curWord );
  END;   

  CloseStream ( bs^.stream, err );
  DISPOSE ( bs );

  END Disconnect;


PROCEDURE EOS
  ( bs : bitStream (* in *) ) : BOOLEAN;

  BEGIN

  IF bs^.streamType = READ THEN
    RETURN ( bs^.curBit = lowBit ) AND Streams.EOS ( bs^.stream );

  ELSE
    RETURN FALSE  (* never get to EOS on write stream ! *)

  END;

  END EOS;


PROCEDURE Read
  ( bs : bitStream (* in/out *) ) : BOOLEAN;

  (* curBit = last bit that has been read *)

  BEGIN
   
  IF bs^.streamType = READWRITE THEN
    Fatal ( 'attempt to read a write bit stream' );

  ELSE WITH bs^ DO
    IF ( curBit = lowBit ) AND ( NOT Streams.EOS ( stream ) ) THEN
      Read16Bit ( stream, curWord );
      curBit := highBit;

    ELSE
      DEC ( curBit );

    END;

    RETURN curBit IN curWord;

    END; (* WITH bs^ *) END; (* IF... ELSE... *)
    
  END Read;


PROCEDURE Write
  ( bs : bitStream; (* in/out *)
     b : BOOLEAN    (* in     *) );

  (* curBit = next bit to write *)

  BEGIN WITH bs^ DO

  IF streamType = READ THEN 
    Fatal ( 'attempt to write a read bit stream' );
  END;
  
  IF b THEN INCL ( curWord, curBit ) END;

  IF curBit = lowBit THEN
    Write16Bit ( stream, curWord ); (* flush full word *)
    curWord := {};
    curBit  := highBit;

  ELSE
   DEC ( curBit ); (* continue to write in current word *)
  
  END; (* IF... ELSE... *)
  
  END; (* WITH bs^ *) 
  END Write;


PROCEDURE ReadChar 
  ( bs : bitStream (* in/out *) ) : CHAR;

  (* Read 8 bits ( most significant first, least last ) and return them 
     as a character. 
  *)

  VAR   i  : charRange;
      char : BITSET;

  BEGIN
  
  char := {};

  FOR i := 7 TO 0 BY -1 DO IF Read ( bs ) THEN INCL ( char, i ) END END;

  RETURN CHAR ( char );

  END ReadChar;


PROCEDURE WriteChar
  ( bs : bitStream; (* in/out *)
     c : CHAR       (* in     *) );

  (* write char as 8 consecutive bits ( most significant first, least 
     last ). 
  *)

  VAR i : charRange;

  BEGIN
  
  FOR i := 7 TO 0 BY -1 DO Write ( bs, i IN BITSET ( c ) ) END;

  END WriteChar;


PROCEDURE ReadLongCard 
  ( bs : bitStream (* in/out *) ) : LONGCARD;

  (* read 32 consecutive bits ( most significant first, least last )
     as a LONGCARD 
  *)

  PROCEDURE ReadCard () : CARDINAL;

    (* read 16 consecutive bits ( most significant first, least last )
       into a bitset and return as a CARDINAL 
    *)

    VAR card : BITSET;
           i : wordRange;
 
    BEGIN
  
    FOR i := highBit TO 0 BY -1 DO
      IF Read ( bs ) THEN
        INCL ( card, i );
      ELSE
        EXCL ( card, i );
      END;
    END;

    RETURN CARDINAL ( card );

    END ReadCard;


  BEGIN

  (* in the expression LONGCARD ( x1 ) + LONGCARD ( x2 ) x2 is 
     apparently evaluated first, making the following structure of
     the RETURN expression NECESSARY !!
  *)

  RETURN LONGCARD ( ReadCard () ) + LONGCARD ( ReadCard () ) * 10000H;

  END ReadLongCard;


PROCEDURE WriteLongCard
  ( bs : bitStream; (* in/out *)
    lc : LONGCARD   (* in     *) );

  (* write lc as 32 consecutive bits ( most significant first, least
     last ).
  *)
  
  PROCEDURE WriteCard
    ( card : CARDINAL (* in *) );

    (* write card as 16 consecutive bits ( most significant first, least
       last ).
    *)

    VAR i : wordRange;

    BEGIN
  
    FOR i := highBit TO 0 BY -1 DO Write ( bs, i IN BITSET ( card ) ) END;

    END WriteCard;

  BEGIN

  WriteCard ( CARDINAL ( lc DIV 10000H ) ); (* high order word *)
  WriteCard ( CARDINAL ( lc MOD 10000H ) ); (* low  order word *)

  END WriteLongCard;

END BitStream.

IMPLEMENTATION MODULE Huffman;

(* File compression algorithm using Huffman coding.
   Based on "Data Compression with Huffman Coding," BYTE March 1986.

   Copyright 1986 by Jonathan Amsterdam.  All Rights Reserved.

   Modified from the original MacModula-2 to TDI Modula-2 for Atari ST

   by Ken Westerback ( 73547,3520 ) August 15, 1986

*)

FROM Storage    IMPORT 
  (* proc *) ALLOCATE, DEALLOCATE;

FROM BitStream  IMPORT
  (* type *) bitStream;

IMPORT BitStream; (* get all procedures as qualified identifiers *)

FROM MyTerminal IMPORT 
  (* proc *) Fatal, WriteLnString;
  
FROM InOut      IMPORT 
  (* proc *) WriteString, WriteCard, WriteLn;


TYPE
    node      = POINTER TO nodeRec;

    nodeRec   = RECORD
                char   : CHAR;
                freq   : LONGCARD;
                child  : ARRAY BOOLEAN OF node;
                parent : node;        (* used for encoding *)
                END;

    huffTree  = POINTER TO htRec;

    htRec     = RECORD
                tree : node;               (* the tree itself *)
                leaf : ARRAY CHAR OF node; (* for encoding    *)
                END;

    charRange = [ 0..255 ]; (* range of ORD ( CHAR ) *)


VAR    tree : ARRAY charRange OF node;  (* temporary list of trees       *)
     nTrees : INTEGER;                  (* index to high tree init to -1 *)


PROCEDURE Huffman
  ( VAR frequency : ARRAY OF LONGCARD (* in *) ) : huffTree;
  
  VAR ht : huffTree;

  BEGIN

  ht := InitHuffTree ( frequency );

  InitTrees ( ht^.leaf );

  WHILE nTrees > 0 DO
    Insert ( CombineNodes ( RemoveSmallest (), RemoveSmallest () ) );
  END;

  ht^.tree := tree[ 0 ];

  RETURN ht;

  END Huffman;


PROCEDURE InitHuffTree 
  ( VAR freq : ARRAY OF LONGCARD (* in *) ) : huffTree;

  VAR  i : charRange;
      ht : huffTree;

  BEGIN
  
  ht := NewHuffTree ();

  FOR i := 0 TO HIGH ( freq ) DO
    IF freq[ i ] # 0 THEN
      ht^.leaf[ CHR ( i ) ] := NewNode ( CHR ( i ), freq[ i ], NIL, NIL );
    END;
  END;
  
  RETURN ht;

  END InitHuffTree;


PROCEDURE InitTrees
  ( VAR leaf : ARRAY OF node (* in *) );

  VAR i : charRange;

  BEGIN

  nTrees := -1;

  FOR i := 0 TO HIGH ( leaf ) DO
    IF leaf[ i ] # NIL THEN Insert ( leaf[ i ] ); END;
  END;

  END InitTrees;


PROCEDURE RemoveSmallest () : node;

  VAR i, smallest : charRange;
     smallestNode : node;

  BEGIN
  
  smallest := 0;

  FOR i := 1 TO nTrees DO
    IF tree[ i ]^.freq < tree[ smallest ]^.freq THEN smallest := i END;
  END;

  smallestNode := tree[ smallest ];
  tree[ smallest ] := tree[ nTrees ];
  DEC ( nTrees );

  RETURN smallestNode;

  END RemoveSmallest;


PROCEDURE Insert
  ( n : node (* in *) );

  BEGIN
  
  INC ( nTrees );
  tree [ nTrees ] := n;

  END Insert;


PROCEDURE WriteCode
  ( bs : bitStream;   (* in/out *) 
    ht : huffTree;    (* in     *)
     c : CHAR         (* in     *) );

  (* Write the code for c onto bs, using ht. By using recursion, we can
     avoid explicitly retracing the path from the root to the leaf. 
  *)

  PROCEDURE WrCode 
    ( n : node (* in *) );

    BEGIN

    IF n^.parent # NIL THEN
      WrCode ( n^.parent );
      BitStream.Write ( bs, n = n^.parent^.child[ TRUE ] );
    END;

    END WrCode;

  BEGIN
  
  IF ht^.leaf[ c ] = NIL THEN
    WriteLnString ( "no code for " ); WriteCard ( CARDINAL ( c ), 5 );
    Fatal ( 'dying' );
  END;

  WrCode ( ht^.leaf[ c ] );

  END WriteCode;


PROCEDURE ReadCode
  ( bs : bitStream;   (* in/out *)
    ht : huffTree     (* in     *) ) : CHAR;

  (* Read bits from bs until a full code is read; return the character *)

  PROCEDURE RdCode 
    ( n:node  (* in *) ) : CHAR;

    BEGIN

    IF Leaf ( n ) THEN
      RETURN n^.char;
    ELSE
      RETURN RdCode ( n^.child[ BitStream.Read ( bs ) ] );
    END;

    END RdCode;

  BEGIN
  
  RETURN RdCode ( ht^.tree );

  END ReadCode;         
            
PROCEDURE WriteTree
  ( bs : bitStream;   (* in/out *) 
    ht : huffTree     (* in     *) );

  (* Write the tree onto the stream.  It is encoded as follows:
     A 1 bit indicates an internal node.
     A 0 bit indicates a leaf; the next 8 bits are the character code. 
     The tree is traversed by preorder traversal: first the root, then
     the left (FALSE) subtree, then the right (TRUE). 
  *)

  PROCEDURE WrTree 
    ( n : node  (* in *) );

    BEGIN

    IF Leaf ( n ) THEN
      BitStream.Write ( bs, FALSE );
      BitStream.WriteChar ( bs, n^.char );
    ELSE
      BitStream.Write ( bs, TRUE );
      WrTree ( n^.child[ FALSE ] );
      WrTree ( n^.child[ TRUE  ] );
    END;

    END WrTree;

  BEGIN
  
  WrTree ( ht^.tree );

  END WriteTree;

PROCEDURE ReadTree
  ( bs : bitStream (* in/out *) ) : huffTree;

  (* Read a huffTree from the stream. See writeTree for the encoding used. 
     Frequency information is NOT preserved.
  *)

  VAR ht : huffTree;

  PROCEDURE RdTree () : node;

    VAR false, true, n : node;

    BEGIN

    IF BitStream.Read ( bs ) THEN (* an internal node *)
      false := RdTree ();
      true  := RdTree ();
      n     := NewNode ( 0C, 0, false, true );
      false^.parent := n;
      true^.parent := n;
      RETURN n;

    ELSE (* a leaf *)
      n := NewNode ( BitStream.ReadChar ( bs ), 0, NIL, NIL );
      ht^.leaf[ n^.char ] := n;
      RETURN n;
    END;
 
    END RdTree;

  BEGIN
  
  ht := NewHuffTree ();
  ht^.tree := RdTree ();

  RETURN ht;

  END ReadTree;


PROCEDURE NewHuffTree () : huffTree;

  VAR  c : CHAR;
      ht : huffTree;

  BEGIN
  
  NEW ( ht );

  FOR c := 0C TO CHR ( HIGH ( ht^.leaf ) ) DO ht^.leaf[ c ] := NIL END;

  RETURN ht;

  END NewHuffTree;
    

PROCEDURE CombineNodes
  ( n1, n2 : node (* in *) ) : node;

  (* used to combine nodes when constructing the coding tree *)

  VAR n : node;

  BEGIN
  
  n := NewNode ( 0C, n1^.freq + n2^.freq, n1, n2 );

  n1^.parent := n;
  n2^.parent := n;

  RETURN n;

  END CombineNodes;


PROCEDURE NewNode 
  (           c : CHAR;      (* in *)
              f : LONGCARD;  (* in *)
    false, true : node       (* in *) ) : node;

  VAR n : node;

  BEGIN
  
  NEW ( n );

  WITH n^ DO
    char           := c;
    freq           := f;
    child[ FALSE ] := false;
    child[ TRUE  ] := true;
    parent         := NIL;
  END;
  
  RETURN n;

  END NewNode;


PROCEDURE FreeNode 
  ( n : node (* in/out *) );

  (* this procedure is not used currently used *)

  BEGIN
  
  IF n # NIL THEN
    FreeNode ( n^.child[ FALSE ] );
    FreeNode ( n^.child[ TRUE  ] );
    DISPOSE ( n );
  END;

  END FreeNode;


PROCEDURE Leaf
  ( n : node   (* in *) ) : BOOLEAN;

  BEGIN
  
  IF n = NIL THEN Fatal ( 'leaf: n NIL' ) END;

  RETURN n^.child[ FALSE ] = NIL;

  END Leaf;


PROCEDURE CodeSize
  ( ht : huffTree;    (* in *)
     c : CHAR         (* in *) ) : CARDINAL;

  (* returns the length of the code for c *)

  VAR i : CARDINAL;
      n : node;

  BEGIN
  
  i := 0;
  n := ht^.leaf[ c ];

  WHILE n # NIL DO INC ( i ); n := n^.parent END;

  RETURN i - 1;

  END CodeSize;


END Huffman.

MODULE Compress;

(* File compression algorithm using Huffman coding.
   Based on "Data Compression with Huffman Coding," BYTE March 1986.

   Copyright 1986 by Jonathan Amsterdam.  All Rights Reserved.

   Modified from the original MacModula-2 to TDI Modula-2 for Atari ST

   by Ken Westerback ( 73547,3520 ) August 15, 1986

*)

FROM Streams    IMPORT 
  (* type *) Stream, StreamKinds, 
  (* proc *) Read8Bit, Write8Bit, OpenStream, CloseStream;

IMPORT Streams; (* get qualified EOS *)

FROM BitStream  IMPORT 
  (* type *) bitStream;

IMPORT BitStream; (* get all procedures as qualified identifiers *)

FROM InOut      IMPORT
  (* proc *) WriteLn, WriteString, ReadString, Write;

FROM LongInOut  IMPORT
  (* proc *) WriteLongCard;

FROM MyTerminal IMPORT 
  (* proc *) WriteLnString, Pause, Fatal;

FROM Huffman    IMPORT 
  (* type *) huffTree, 
  (* proc *) Huffman, CodeSize,
             ReadCode, WriteCode, 
             ReadTree, WriteTree; 

FROM Strings    IMPORT 
  (* type *) String,
  (* proc *) Length, Assign;

FROM RealInOut  IMPORT 
  (* proc *) WriteReal;  


VAR frequency : ARRAY CHAR OF LONGCARD;
    fileSize  : LONGCARD;
    hTree     : huffTree;     

    inFileName, outFileName : String;


PROCEDURE DoFreq;

  (* Obtain frequency count from file *)
  
  VAR cs : Stream;
     err : INTEGER;

  BEGIN
  
  OpenStream ( cs, inFileName, READ, err);        (* read file *)

  IF err = 0 THEN (* ok to use *)
    FreqCount ( cs );
    CloseStream ( cs, err );

  ELSE
    Fatal ( ">> could not open input file" );

  END;

  END DoFreq;

PROCEDURE FreqCount
  ( cs : Stream (* in/out *) );

  VAR c : CHAR;

  BEGIN
  
  FOR c := 0C TO CHR ( HIGH ( frequency ) ) DO frequency[ c ] := 0 END;

  Read8Bit ( cs, c );
 
  WHILE NOT Streams.EOS ( cs ) DO
    INC ( frequency[ c ] );
    INC ( fileSize );
    Read8Bit ( cs, c );
  END;
  
  INC ( frequency[ c ] );
  INC ( fileSize );

  WriteLn;
  WriteLnString ( '--- Character frequency table constructed' );

  END FreqCount;


PROCEDURE DoOutput;

  (* Output encoded file *)
  
  VAR  inStream : Stream;
      outStream : bitStream;
              c : CHAR;
            err : INTEGER;
      charCount : CARDINAL;

  BEGIN

  err := 0;

  OpenStream ( inStream, inFileName, READ, err );
  IF err # 0 THEN
    Fatal ( '---could not connect to input stream' )
  END;

  BitStream.OpenBitStream ( outStream, outFileName, READWRITE, err );
  IF err # 0 THEN 
    Fatal ( '--- could not connect to output stream' )
  END;

  BitStream.WriteLongCard ( outStream, fileSize );
  WriteTree ( outStream, hTree );
  WriteLnString ( '--- Huffman Encoding Tree written to output' );
  WriteLnString ( '--- Writing encoded data' );
  WriteLn;

  charCount := 0;
  Read8Bit ( inStream, c );
  WHILE NOT Streams.EOS ( inStream ) DO
    WriteCode ( outStream, hTree, c );
    INC ( charCount );
    IF charCount = 100 THEN Write ( '.' ); charCount := 0 END;
    Read8Bit ( inStream, c );
  END;
 
  WriteCode ( outStream, hTree, c );
  CloseStream ( inStream, err );
  BitStream.Disconnect ( outStream );

  END DoOutput;


PROCEDURE ComputeStats;

  (* Compute statistics on how much space was saved *)

  VAR                        c : CHAR;
    origBits, compBits, nChars : LONGCARD;
                       savings : REAL;
    realCompBits, realOrigBits : REAL;

  BEGIN
  
  origBits := fileSize * 8;
  compBits := 0;
  nChars   := 0;

  FOR c := 0C TO CHR ( HIGH ( frequency ) ) DO
    IF frequency[ c ] # 0 THEN
      INC ( nChars );
      compBits := compBits + 
                     LONGCARD ( CodeSize ( hTree, c ) ) * frequency[ c ];
    END;
  END;
 
  WriteLnString ( "number of different characters: " );
    WriteLongCard ( nChars, 12 );
  WriteLnString ( "original file size ( bits ) : ");
    WriteLongCard ( origBits, 12 );
  WriteLnString ( "compressed file size ( bits ) : ");
    WriteLongCard ( compBits, 12 ); 
  WriteLnString ( "percent savings: " );
    realCompBits := FLOAT ( CARDINAL ( compBits DIV 10000H ) ) * 65536.0
                    +
                    FLOAT ( CARDINAL ( compBits MOD 10000H ) );
    realOrigBits := FLOAT ( CARDINAL ( origBits DIV 10000H ) ) * 65536.0
                    +
                    FLOAT ( CARDINAL ( origBits MOD 10000H ) );
    savings := ( 1.0 - ( realCompBits / realOrigBits ) ) * 100.0;
    WriteReal ( savings, 5 );
  WriteLnString ( "compressed size, including bookkeeping: " );
    (* add 16 bits for character, count, 10n-1 bits for tree *)
    INC ( compBits, 16 + 10 * nChars - 1 );
    WriteLongCard ( compBits, 5 ); 
  WriteLnString ( "true percent savings: " );
    realCompBits := FLOAT ( CARDINAL ( compBits DIV 10000H ) ) * 65536.0
                    +
                    FLOAT ( CARDINAL ( compBits MOD 10000H ) );
    savings := ( 1.0 - ( realCompBits / realOrigBits ) ) * 100.0;
    WriteReal ( savings, 5 );

  END ComputeStats;

MODULE Uncompress;

(* File compression algorithm using Huffman coding.
   Based on "Data Compression with Huffman Coding," BYTE March 1986.

   Copyright 1986 by Jonathan Amsterdam.  All Rights Reserved.

   Modified from the original MacModula-2 to TDI Modula-2 for Atari ST

   by Ken Westerback ( 73547,3520 ) August 15, 1986

*)

FROM Streams     IMPORT
  (* type *) Stream, StreamKinds, 
  (* proc *) OpenStream, CloseStream, Read8Bit, Write8Bit; 

IMPORT Streams; (* get qualified EOS *)

FROM BitStream   IMPORT 
  (* type *) bitStream;

IMPORT BitStream; (* get all procedures as qualified identifiers *)

FROM InOut       IMPORT 
  (* proc *) ReadString, WriteLn, WriteString, Write;

FROM LongInOut   IMPORT
  (* proc *) WriteLongCard;
 
FROM MyTerminal  IMPORT 
  (* proc *) Pause, WriteLnString, Fatal;

FROM Huffman     IMPORT 
  (* type *) huffTree, 
  (* proc *) Huffman, ReadCode, ReadTree;

FROM Strings     IMPORT 
  (* type *) String,
  (* proc *) Length, Assign;


VAR inFileName, outFileName: String;


PROCEDURE DoUncompress;

  VAR  inStream : bitStream;
      outStream : Stream;
            err : INTEGER;
    fileSize, i : LONGCARD;  (* number of characters in file *)
          hTree : huffTree;
      charCount : CARDINAL;

  BEGIN
 
  err := 0;
  
  BitStream.OpenBitStream ( inStream, inFileName, READ, err );
  IF err # 0 THEN
    Fatal ( '--- could not connect to input stream' )
  END;

  OpenStream ( outStream, outFileName, READWRITE, err);
  IF err # 0 THEN
    Fatal ( '--- could not connect to output stream' )
  END;

  fileSize := BitStream.ReadLongCard ( inStream );

  WriteLnString ( 'number of encoded characters = ' );
  WriteLongCard ( fileSize, 12 );

  hTree    := ReadTree ( inStream );
  WriteLnString ( 'coding tree successfully read...starting on data' );
  WriteLn;

  charCount := 0;
  FOR i := 1 TO fileSize DO
    Write8Bit ( outStream, ReadCode ( inStream, hTree ) );
    INC ( charCount );
    IF charCount = 100 THEN Write ( '.' ); charCount := 0 END; 
  END;

  CloseStream ( outStream, err );
  BitStream.Disconnect ( inStream );

  END DoUncompress;


PROCEDURE DoFileNames;

  VAR len : CARDINAL;

  BEGIN
  
  len := Length ( inFileName );
  Assign ( outFileName, inFileName );
  outFileName [ len-1 ] := 'u';

  WriteLn;
  WriteLnString ( 'Input file  = ' ); WriteString ( inFileName );

  WriteLn;
  WriteLnString ( 'Output file = ' ); WriteString ( outFileName );

  END DoFileNames;



BEGIN
    
WriteLnString ( 'Huffman De-coding program');

WriteLnString ( 'Enter input file name : ' );
ReadString ( inFileName );
    
DoFileNames;
    
DoUncompress;
    
Pause ( 'De-coding complete - press any key to exit' );

END Uncompress.
